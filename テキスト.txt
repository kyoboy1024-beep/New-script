    local function setThermalLighting()
        originalSettings.Brightness = Lighting.Brightness
        originalSettings.Ambient = Lighting.Ambient
        originalSettings.ColorShift_Top = Lighting.ColorShift_Top
        originalSettings.ColorShift_Bottom = Lighting.ColorShift_Bottom

        originalSettings.FogColor = Lighting.FogColor
        originalSettings.GlobalShadows = Lighting.GlobalShadows

        Lighting.FogColor = Color3.fromRGB(40, 20, 60)
        Lighting.GlobalShadows = false
    end

    local function resetLighting()
        if originalSettings.Brightness then Lighting.Brightness = originalSettings.Brightness end
        if originalSettings.Ambient then Lighting.Ambient = originalSettings.Ambient end
        if originalSettings.ColorShift_Top then Lighting.ColorShift_Top = originalSettings.ColorShift_Top end
        if originalSettings.ColorShift_Bottom then Lighting.ColorShift_Bottom = originalSettings.ColorShift_Bottom end
        if originalSettings.OutdoorAmbient then Lighting.OutdoorAmbient = originalSettings.OutdoorAmbient end

        if originalSettings.GlobalShadows ~= nil then Lighting.GlobalShadows = originalSettings.GlobalShadows end
    end

    local function addPlayerHighlight(player)
        if not player.Character then return end
        
        local highlight = Instance.new("Highlight")
        highlight.Name = "ThermalHighlight"
        highlight.FillColor = Color3.fromRGB(255, 140, 50)
        highlight.OutlineColor = Color3.fromRGB(255, 200, 100)
        highlight.FillTransparency = 0.3
        highlight.OutlineTransparency = 0.1
        highlight.Parent = player.Character
        playerHighlights[player] = highlight
    end

    local function removePlayerHighlight(player)
        if playerHighlights[player] then
            pcall(function() playerHighlights[player]:Destroy() end)
            playerHighlights[player] = nil
        end
    end

    local function updateAllPlayerHighlights()
        for _, player in ipairs(Players:GetPlayers()) do
            if player ~= me then
                if player.Character then
                    if not playerHighlights[player] then
                        addPlayerHighlight(player)
                    end
                else
                    removePlayerHighlight(player)
                end
            end
        end
    end

    local function clearAllHighlights()
        for _, highlight in pairs(playerHighlights) do
            pcall(function() highlight:Destroy() end)
        end
        playerHighlights = {}
    end

        if thermalEnabled then return end
        thermalEnabled = true
        
        setThermalLighting()
        updateAllPlayerHighlights()
        
        thermalConnection = RunService.Heartbeat:Connect(function()
            if not thermalEnabled then return end
            if tick() % 2 < 0.033 then
                updateAllPlayerHighlights()
            end
        end)

            if thermalEnabled and player ~= me then
                player.CharacterAdded:Connect(function()
                    task.wait(0.5)
                    if thermalEnabled then
                        addPlayerHighlight(player)
                    end
                end)
                if player.Character then
                    task.wait(0.5)
                    if thermalEnabled then
                        addPlayerHighlight(player)
                    end
                end
            end

            if player ~= me then
                player.CharacterAdded:Connect(function()
                    task.wait(0.5)
                    if thermalEnabled then
                        addPlayerHighlight(player)
                    end
                end)
            end
        end
    end

    local function createBodyMovers(part)
        local bp = part:FindFirstChild("BP") or Instance.new("BodyPosition")
        bp.Name = "BP"
        bp.P = 20000
        bp.D = 200
        bp.MaxForce = Vector3.new(5e6, 5e6, 5e6)

        bg.Name = "BG"
        bg.P = 20000
        bg.D = 200
        bg.MaxTorque = Vector3.new(5e6, 5e6, 5e6)
        bg.CFrame = part.CFrame

        part.Anchored = false
    end

    local function enableCollision(model)
        for _, p in pairs(model:GetDescendants()) do
            if p:IsA("BasePart") then
                if originalCollides[p] == nil then
                    originalCollides[p] = p.CanCollide
                end
                p.CanCollide = originalCollides[p]
            end
        end
    end

    local function collectModelsInFolder(folder)
        for _, toy in ipairs(folder:GetDescendants()) do
            if toy:IsA("Model") then
                local part = toy:FindFirstChildWhichIsA("BasePart")
                if part and not activeToys[toy] then
                    activeToys[toy] = {
                        part = part,
                        index = math.random(1, 1000)
                    }
                    enableCollision(toy)
                    createBodyMovers(part)
                end
            end
        end
    end

    local function addToysToAura(target)

            local meFolder = workspace:FindFirstChild(me.Name .. "SpawnedInToys")
            if meFolder then
                collectModelsInFolder(meFolder)
            end

            local plr = Players.LocalPlayer
            local folders = {}

            local plotsFolder = workspace:FindFirstChild("Plots")
            if plotsFolder then
                for i = 1, 5 do
                    local plot = plotsFolder:FindFirstChild("Plot" .. i)
                    if plot then
                        local ownersFolder = plot:FindFirstChild("PlotSign") and plot.PlotSign:FindFirstChild("ThisPlotsOwners")
                        if ownersFolder then
                            for _, v in ipairs(ownersFolder:GetChildren()) do
                                if v:IsA("ValueBase") and v.Value == plr.Name then
                                    local plotItemsFolder = workspace:FindFirstChild("PlotItems")
                                    if plotItemsFolder and plotItemsFolder:FindFirstChild(plot.Name) then
                                        table.insert(folders, plotItemsFolder:FindFirstChild(plot.Name))
                                    end
                                end
                            end
                        end
                    end
                end
            end

            for _, folder in ipairs(folders) do
                collectModelsInFolder(folder)
            end

        elseif target == "allobject" then
            for _, player in ipairs(Players:GetPlayers()) do
                local playerFolder = workspace:FindFirstChild(player.Name .. "SpawnedInToys")
                if playerFolder then
                    collectModelsInFolder(playerFolder)
                end

                "Glass Boxes", "Ladders", "Large Animals", "Music", "NOT New",

                    if folderplot then

        for toy, _ in pairs(activeToys) do

    local function spawnMissileAtPlayer()
        local hrp = me.Character and me.Character:FindFirstChild("HumanoidRootPart")
        if not hrp then return false end
        
        local charCF = hrp.CFrame
        local forwardDirection = charCF.LookVector
        local spawnPos = hrp.Position + forwardDirection * 1 + Vector3.new(0, 2, 0)
        local spawnCF = CFrame.new(spawnPos, spawnPos + forwardDirection)
        
        local success, missile = pcall(function()
            return ReplicatedStorage.MenuToys.SpawnToyRemoteFunction:InvokeServer(
                "BombMissile",
                spawnCF,
                Vector3.zero
            )
        end)
        
        if success and missile then
            local part = missile:FindFirstChildWhichIsA("BasePart")
            if part then
                activeToys[missile] = {
                    part = part,
                    index = math.random(1, 1000)
                }
                enableCollision(missile)
                createBodyMovers(part)
            end
            return true
        end
        return false
    end

    local autoReloadLoop = nil

    local function startAutoReload()
        if autoReloadLoop then return end
        autoReloadRunning = true
        
        autoReloadLoop = task.spawn(function()
            while autoReloadRunning and ICBMSettings.AutoReload and toys.mod == "BombMissile" and toys.isMoving do
                local hasMissile = false
                for toy, _ in pairs(activeToys) do
                    if toy and toy.Name == "BombMissile" and toy.Parent then
                        hasMissile = true
                        break
                    end
                end
                
                if not hasMissile then
                    task.wait(0.3)
                    local success = spawnMissileAtPlayer()
                    if success then
                        bombMissileStartTime = nil
                    end
                end
                
                task.wait(0.3)
            end
            autoReloadLoop = nil
        end)
    end

    local function stopAutoReload()
        autoReloadRunning = false
        if autoReloadLoop then
            task.cancel(autoReloadLoop)
            autoReloadLoop = nil
        end
    end

    local function bombMissileEffect()
        local plr = Players.LocalPlayer
        local camera = workspace.CurrentCamera
        
        local missile = nil
        for toy, data in pairs(activeToys) do
            if toy.Name == "BombMissile" then
                missile = toy
                break
            end
        end
        
        if not missile then
            if originalCameraSubject then
                camera.CameraSubject = originalCameraSubject
                camera.CameraType = originalCameraType
                originalCameraSubject = nil
                bombMissileStartTime = nil
            end
            return
        end
        
        local part = missile:FindFirstChildWhichIsA("BasePart")
        if not part or not part:FindFirstChild("BP") or not part:FindFirstChild("BG") then return end
        
        if not bombMissileStartTime then
            bombMissileStartTime = os.clock()
            originalCameraSubject = camera.CameraSubject
            originalCameraType = camera.CameraType
            camera.CameraSubject = part
        end
        
        local elapsed = os.clock() - bombMissileStartTime

        if elapsed < ICBMSettings.AscendTime then
            part.BP.Position = part.Position + Vector3.new(0, ICBMSettings.AscendSpeed, 0)
        else
            part.BP.Position = part.Position + camera.CFrame.LookVector * ICBMSettings.FlySpeed
            part.BG.CFrame = camera.CFrame * CFrame.Angles(math.rad(toys.missileRotationOffset), 0, 0)
        end
    end

    local function helicopterEffect()
        local plr = Players.LocalPlayer
        local camera = workspace.CurrentCamera
        local heli = nil
        for toy, data in pairs(activeToys) do
            if toy.Name == "FlyingToyHelicopter" then
                heli = toy
                break
            end
        end

        if not heli then
            if originalCameraSubject then
                camera.CameraSubject = originalCameraSubject
                camera.CameraType = originalCameraType
                originalCameraSubject = nil
                if plr.Character and plr.Character:FindFirstChild("HumanoidRootPart") then
                    plr.Character.HumanoidRootPart.Anchored = false
                end
            end
            return
        end

        local part = heli:FindFirstChildWhichIsA("BasePart")
        if not part or not part:FindFirstChild("BP") or not part:FindFirstChild("BG") then return end

        if not originalCameraSubject then
            originalCameraSubject = camera.CameraSubject
            originalCameraType = camera.CameraType
            local cameraTarget = heli:FindFirstChild("Body") or part
            camera.CameraSubject = cameraTarget
            if plr.Character and plr.Character:FindFirstChild("HumanoidRootPart") then
                plr.Character.HumanoidRootPart.Anchored = true
            end
        end

        local moveSpeed = 0.3
        local rotateSpeed = 1.5
        local verticalSpeed = 0.3

        local moveDir = Vector3.new(0, 0, 0)
        local rotationChange = 0

        if keysDown[Enum.KeyCode.W] then
            moveDir = moveDir - part.CFrame.LookVector * moveSpeed
        end
        if keysDown[Enum.KeyCode.S] then
            moveDir = moveDir + part.CFrame.LookVector * moveSpeed
        end
        if keysDown[Enum.KeyCode.A] then
            rotationChange = rotationChange + rotateSpeed
        end
        if keysDown[Enum.KeyCode.D] then
            rotationChange = rotationChange - rotateSpeed
        end
        if keysDown[Enum.KeyCode.Z] then
            moveDir = moveDir + Vector3.new(0, verticalSpeed, 0)
        end
        if keysDown[Enum.KeyCode.X] then
            moveDir = moveDir - Vector3.new(0, verticalSpeed, 0)
        end

        if moveDir.Magnitude > 0 then
            part.BP.Position = part.Position + moveDir
        end

        if rotationChange ~= 0 then
            local x, y, z = part.BG.CFrame:ToEulerAnglesYXZ()
            part.BG.CFrame = CFrame.fromEulerAnglesYXZ(x, y + math.rad(rotationChange), z)
        end
    end

    local function combinedEffect()
        if toys.mod == "BombMissile" then
            bombMissileEffect()
        elseif toys.mod == "Helicopter" then
            helicopterEffect()
        end
    end

    function toggleEffect()
        if toys.isMoving then
            addToysToAura(selectedTarget)

            if effectCoroutine then
                coroutine.close(effectCoroutine)
            end

            effectCoroutine = coroutine.create(function()
                while toys.isMoving do
                    combinedEffect()
                    task.wait()
                end
            end)
            coroutine.resume(effectCoroutine)
            
            if ICBMSettings.AutoReload then
                startAutoReload()
            end
        else
            if effectCoroutine then
                coroutine.close(effectCoroutine)
                effectCoroutine = nil
            end
            stopAutoReload()
            clearEffects()
            
            if originalCameraSubject then
                local camera = workspace.CurrentCamera
                camera.CameraSubject = originalCameraSubject
                camera.CameraType = originalCameraType
                originalCameraSubject = nil
                bombMissileStartTime = nil
                
                local plr = Players.LocalPlayer
                if plr.Character and plr.Character:FindFirstChild("HumanoidRootPart") then
                    plr.Character.HumanoidRootPart.Anchored = false
                end
            end
        end
    end

        Name = "ICBM Settings"
    })

    ToyModTab:AddSlider({
        Name = "Ascend Speed",
        Min = 0.5,
        Max = 10,
        Default = 1,
        Increment = 0.5,
        Callback = function(value)
            ICBMSettings.AscendSpeed = value
        end
    })

    ToyModTab:AddSlider({
        Name = "Ascend Time (seconds)",
        Min = 0.5,
        Max = 10,
        Default = 1,
        Increment = 0.5,
        Callback = function(value)
            ICBMSettings.AscendTime = value
        end
    })

    ToyModTab:AddSlider({
        Name = "Fly Speed",
        Min = 1,
        Max = 20,
        Default = 2,
        Increment = 1,
        Callback = function(value)
            ICBMSettings.FlySpeed = value
        end
    })

    ToyModTab:AddToggle({
        Name = "Auto Reload (自動補充)",
        Default = false,
        Save = true,
        Callback = function(state)
            ICBMSettings.AutoReload = state
            if toys.isMoving and toys.mod == "BombMissile" then
                if state then
                    startAutoReload()
                else
                    stopAutoReload()
                end
            end
        end
    })

    ToyModTab:AddToggle({
        Name = "ICBM",
        Default = false,
        Save = true,
        Callback = function(state)
            toys.isMoving = state
            toys.mod = state and "BombMissile" or nil
            toggleEffect()
        end
    })

    ToyModTab:AddToggle({
        Name = "Third Person",
        Default = false,
        Callback = function(state)
            if state then
                me.CameraMode = Enum.CameraMode.Classic
                me.CameraMaxZoomDistance = 100
                me.CameraMinZoomDistance = 0.5
            else
                me.CameraMode = Enum.CameraMode.LockFirstPerson
                me.CameraMaxZoomDistance = 0.5
                me.CameraMinZoomDistance = 0.5
            end
        end    
    })

    ToyModTab:AddToggle({
        Name = "Helicopter Control",
        Default = false,
        Save = true,
        Callback = function(state)
            toys.isMoving = state
            toys.mod = state and "Helicopter" or nil
            toggleEffect()
        end
    })

    ToyModTab:AddDropdown({
        Name = "Toy Mode",
        Default = target,
        Options = {"meobject"},
        Callback = function(sel)
            selectedTarget = sel
            addToysToAura(selectedTarget)
        end
    })

    OrionLib:Init()
